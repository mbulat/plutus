def self.balance
  accounts_balance = BigDecimal.new('0')
  accounts = self.all
  accounts.each do |revenue|
    unless revenue.contra
      accounts_balance += revenue.balance
    else
      accounts_balance -= revenue.balance
    end
  end
  accounts_balance
end
